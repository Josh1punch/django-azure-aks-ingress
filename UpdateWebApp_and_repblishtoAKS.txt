
#create updated image docker container
docker build -f Dockerfile -t django-aks:v1.4.1 .
#tag this to azure acr (registry)
 docker tag django-aks:v1.4.1 djangoaksregistry.azurecr.io/django-aks:v1.4.1

#login to my ask registry first
az acr login --name djangoaksregistry

#push it to AKS
docker push djangoaksregistry.azurecr.io/django-aks:v1.4.1

#use kubtl to update the webapp on AKS
kubectl set image deployment django-web-app django-web-container=djangoaksregistry.azurecr.io/django-aks:v1.4.1 --namespace djaks
